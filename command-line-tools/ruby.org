#+TITLE: Ruby
#+begin_src sh
brew install ruby
#+end_src

Add brew ruby path to zshrc
#+begin_src sh
echo 'export PATH="/usr/local/opt/ruby/bin:$PATH"' >> ~/.zshrc
#+end_src

#+begin_src sh
gem install bundler jekyll
